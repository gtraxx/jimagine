Plugin nicenotify

Le plugin nicenotify permet d'effectuer des requêtes ajax avec des helpers pour avoir une notification simple

Affiche une notification simple sans requête ajax

$.nicenotify({ ntype: "simple" });

Affiche une notification avec une requête ajax sans paramètres

$.nicenotify({ ntype: "ajax", uri: '', typesend: 'get', datatype: 'json' });

Affiche une notification avec une requête ajax avec paramètres

                $.nicenotify({
                    ntype: "submit",
                    uri: '',
                    typesend: 'post',
                    idforms: form,
                    resetform:true,
                    beforeParams:function(){
                        $(":submit").button({
                            disabled: true,
                            label: "Please wait...."
                        });
                    },
                    successParams:function(e){
                        $(":submit").button({disabled: false,label: "Envoyer"});
                        $.nicenotify.initbox(e,{
                            display: true,
                            refresh: false,
                            delay: 3000,
                            text: ''
                        },{
                            box:'meerkat',
                            time: 4,
                            elemid : '#notify-header',
                            elemclass : '.mc-head-request',
                            background : "#efefef"
                        });
                    }
                });
